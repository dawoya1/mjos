快速开始

1. 安装

npm install mjos

2. 使用

const { MJOS } = require('mjos');

const mjos = new MJOS();
await mjos.start();

// 存储记忆
const memoryId = mjos.remember('重要信息', ['标签'], 0.8);

// 创建任务
const taskId = mjos.createTask('任务标题', '任务描述', 'high');

// 分配任务
mjos.assignTask(taskId, 'moxiaozhi');

await mjos.stop();
