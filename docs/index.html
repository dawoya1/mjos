Quick Start

# Global installation
npm install -g mjos@2.1.11

# Or use npx
npx mjos@2.1.11 --help
# Start MCP server
npx mjos@2.1.11 mcp-server

Add to .cursor/mcp.json:

{
  "mcpServers": {
    "mjos": {
      "command": "npx",
      "args": ["-y", "mjos@2.1.11", "mcp-server"],
      "env": {"MJOS_LOG_LEVEL": "info"}
    }
  }
}
import { MJOS } from 'mjos';

const mjos = new MJOS();
await mjos.start();

// Store memory
await mjos.memory.store('project-info', 'MJOS system overview', {
  tags: ['project', 'info'],
  importance: 8
});

// Query memories
const memories = await mjos.memory.query({ tags: ['project'] });

await mjos.stop();
